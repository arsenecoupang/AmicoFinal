작업할 것
1. 12시 자동 리셋(문제 생성, 채팅방 삭제, MVP 투표)

보완할 것
- MVP 투표 중복 방지(한 번만 투표 가능)
- MVP 실명 노출 UI/보안
- MVP 히스토리 UI 개선
- 투표 결과 실시간 반영

--- Supabase 테이블 생성 쿼리 예시 ---
profiles: id(uuid, PK), email, username, realname, temperature(int)
questions: id(uuid, PK), question1, question2, option1, option2, created_at
rooms: id(uuid, PK), question_id, option, members(json)
messages: id(uuid, PK), room_id, user_id, content, created_at
votes: id(uuid, PK), room_id, voter_id, candidate_id
mvp_history: id(uuid, PK), room_id, mvp_id, realname, date

-- SQL 예시 --
CREATE TABLE profiles (
	id uuid PRIMARY KEY,
	email text,
	username text,
	realname text,
	temperature int
);
CREATE TABLE questions (
	id uuid PRIMARY KEY,
	question1 text,
	question2 text,
	option1 text,
	option2 text,
	created_at timestamp with time zone default now()
);
CREATE TABLE rooms (
	id uuid PRIMARY KEY,
	question_id uuid references questions(id),
	option text,
	members jsonb
);
CREATE TABLE messages (
	id uuid PRIMARY KEY,
	room_id uuid references rooms(id),
	user_id uuid references profiles(id),
	content text,
	created_at timestamp with time zone default now()
);
CREATE TABLE votes (
	id uuid PRIMARY KEY,
	room_id uuid references rooms(id),
	voter_id uuid references profiles(id),
	candidate_id uuid references profiles(id)
);
CREATE TABLE mvp_history (
	id uuid PRIMARY KEY,
	room_id uuid references rooms(id),
	mvp_id uuid references profiles(id),
	realname text,
	date date
);
